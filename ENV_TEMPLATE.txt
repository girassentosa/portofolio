# ============================================
# TEMPLATE FILE .env.local
# ============================================
# Copy isi file ini ke file bernama: .env.local
# Letakkan di root project (folder yang sama dengan package.json)

# ============================================
# UNTUK DEVELOPMENT LOKAL (XAMPP):
# ============================================

DB_HOST=localhost
DB_USER=root
DB_PASSWORD=
DB_NAME=portfolio_db


# ============================================
# UNTUK PRODUCTION (INFINITYFREE):
# ============================================
# Ganti dengan kredensial ASLI dari cPanel InfinityFree:

# DB_HOST=localhost
# DB_USER=id12345_portfolio_user
# DB_PASSWORD=password_kuat_anda_123
# DB_NAME=id12345_portfolio_db


# ============================================
# CARA MEMBUAT FILE .env.local:
# ============================================

## DI LOKAL (Windows):
## 1. Buat file baru bernama: .env.local
## 2. Copy 4 baris DB_HOST sampai DB_NAME (untuk XAMPP)
## 3. Paste ke file .env.local
## 4. Save

## DI INFINITYFREE (Server):
## 1. Login ke cPanel InfinityFree
## 2. Buka File Manager
## 3. Navigate ke folder project Anda (htdocs atau public_html)
## 4. Klik "+ File" atau "New File"
## 5. Nama file: .env.local
## 6. Klik "Edit" pada file tersebut
## 7. Copy 4 baris DB_HOST sampai DB_NAME (untuk InfinityFree)
## 8. GANTI dengan kredensial ASLI dari cPanel:
##    - DB_USER: Username MySQL (lihat di MySQL Databases)
##    - DB_PASSWORD: Password yang Anda buat saat create user MySQL
##    - DB_NAME: Nama database (lihat di MySQL Databases)
## 9. Save Changes


# ============================================
# CONTOH KREDENSIAL INFINITYFREE:
# ============================================
# Setelah Anda membuat database di cPanel, kredensialnya akan seperti:

# DB_HOST=localhost
# DB_USER=epiz_12345678_portfolio
# DB_PASSWORD=MyStrongPassword123!
# DB_NAME=epiz_12345678_portfolio_db

# CATATAN:
# - Prefix "epiz_" atau "id" adalah ID account InfinityFree Anda
# - Username dan database name biasanya diawali dengan prefix tersebut
# - Password adalah yang Anda buat sendiri saat create user


# ============================================
# TROUBLESHOOTING:
# ============================================
# Jika error "Access denied for user":
# - Cek DB_USER apakah benar (lihat di cPanel MySQL Databases)
# - Cek DB_PASSWORD apakah benar
# - Pastikan user sudah di-assign ke database dengan ALL PRIVILEGES

# Jika error "Unknown database":
# - Cek DB_NAME apakah benar (lihat di cPanel MySQL Databases)
# - Pastikan database sudah dibuat dan sudah di-import SQL file
